plot_item_selections <- function(d) 
{
  
  d %>% arrange(found) %>%
    mutate(class = as_factor(class)) %>%
    ggplot(aes(x, y, colour = class, label = found)) + 
    geom_point(size = 3) + geom_label() + 
    geom_path(aes( group = 1), colour = "black")
  
}
